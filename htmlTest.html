Sample

<!DOCTYPE html>
<html>
  <head>
    <title>Basic HTML</title>
  </head>
  <body>
  <!--HTML-->
  <p id="text"></p>
  <ol>
    <li>Do item 1</li>
    <li>Do item 2</li>
    <li>Do item 3</li>
  </ol>
  <!--Javascript-->
  <script>
    var alerts = ["Hi","Bye"];
    for(var i = 0; i < alerts.length; i++){
      /*Alert each item in alerts list*/
      alert(alerts[i]);
      document.getElementById("text").innerHTML += alerts[i];
    }
  </script>
</body>
</body>
</html>

<!DOCTYPE html>

  </head>

  </body>
